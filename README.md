```

        "roll_no": 1,

        "gender": "M",
        "dob": "01/01/2001",
        "contact_no.": "+91xxxxxxxxx",
        "city": "Delhi",
        "subjects": [
            {
                "sub_id": 1,
                "marks": 92
            },
            {
                "sub_id": 2,
                "marks": 86
            }
        ]
    }
```
